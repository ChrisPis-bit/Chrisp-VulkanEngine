cmake_minimum_required(VERSION "3.19.2")

add_subdirectory("libraries/glfw-3.4")

project(ChrispEngine)
set(CMAKE_CXX_STANDARD 17)

file(GLOB_RECURSE MY_SOURCES CONFIGURE_DEPENDS "${CMAKE_CURRENT_SOURCE_DIR}/src/*.cpp")
add_executable(${CMAKE_PROJECT_NAME} "${MY_SOURCES}")

target_compile_definitions("${CMAKE_PROJECT_NAME}" PUBLIC PROJECT_PATH="${CMAKE_CURRENT_SOURCE_DIR}/src/") # Uncomment this line to setup for development
#target_compile_definitions("${CMAKE_PROJECT_NAME}" PUBLIC PROJECT_PATH="./shaders") # Uncomment this line to setup for production

find_package(Vulkan REQUIRED)

target_include_directories(ChrispEngine PUBLIC "${CMAKE_CURRENT_SOURCE_DIR}/include/")

target_link_libraries(ChrispEngine PRIVATE glfw Vulkan::Vulkan)
